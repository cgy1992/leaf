require 'leaf.rect' require 'leaf.vector' local vector = leaf.vector local rect = leaf.rect local layout = {}

Return the position for an element of a list

layout.list = function(rect, position, conf) local conf = conf or {} local margin = conf.margin or 1 local spacing = conf.spacing or 1 local x = margin local y = margin + (position - 1) * spacing return vector.translate(x, y, rect.left, rect.top) end

Export

leaf.layout = layout
