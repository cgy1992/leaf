local function new(class, ...) local obj = setmetatable({}, class) if obj.init then obj:init(...) end return obj end local function extend(class, sub) local sub = sub or {} sub.__index = sub sub.__call = new return setmetatable(sub, class) end local Object = extend({}, {extend=extend})

Export

leaf.Object = Object
